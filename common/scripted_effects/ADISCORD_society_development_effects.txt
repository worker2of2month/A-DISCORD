# Generic society development effects.
# Use variables for levels and ideas for visible national spirit effects.

ADISCORD_apply_society_development_level = {
	if = {
		limit = { NOT = { has_variable = ADISCORD_society_development_level } }
		set_variable = { var = ADISCORD_society_development_level value = 1 }
	}
	if = {
		limit = { ADISCORD_society_development_at_most_1 = yes }
		ADISCORD_set_society_development_1 = yes
	}
	else_if = {
		limit = { ADISCORD_society_development_exact_2 = yes }
		ADISCORD_set_society_development_2 = yes
	}
	else_if = {
		limit = { ADISCORD_society_development_exact_3 = yes }
		ADISCORD_set_society_development_3 = yes
	}
	else_if = {
		limit = { ADISCORD_society_development_exact_4 = yes }
		ADISCORD_set_society_development_4 = yes
	}
	else = {
		ADISCORD_set_society_development_5 = yes
	}
}

ADISCORD_set_society_development_1 = {
	set_variable = { var = ADISCORD_society_development_level value = 1 }
	set_variable = { var = ADISCORD_society_development_progress value = 0 }
	remove_ideas = ADISCORD_society_development_1
	remove_ideas = ADISCORD_society_development_2
	remove_ideas = ADISCORD_society_development_3
	remove_ideas = ADISCORD_society_development_4
	remove_ideas = ADISCORD_society_development_5
	add_ideas = ADISCORD_society_development_1
}

ADISCORD_update_total_country_development = {
	ADISCORD_initialize_society_development_variables = yes

	set_variable = { var = ADISCORD_total_development_score value = 0 }
	add_to_variable = { var = ADISCORD_total_development_score value = ADISCORD_society_development_level }
	add_to_variable = { var = ADISCORD_total_development_score value = ADISCORD_social_system_development_level }
	add_to_variable = { var = ADISCORD_total_development_score value = ADISCORD_army_development_level }
	add_to_variable = { var = ADISCORD_total_development_score value = ADISCORD_cultural_development_level }
	add_to_variable = { var = ADISCORD_total_development_score value = ADISCORD_state_development_level }
	add_to_variable = { var = ADISCORD_total_development_score value = ADISCORD_economic_development_level }

	if = {
		limit = { check_variable = { var = ADISCORD_total_development_score value = 30 compare = greater_than_or_equals } }
		set_variable = { var = ADISCORD_total_development_level value = 5 }
	}
	else_if = {
		limit = { check_variable = { var = ADISCORD_total_development_score value = 24 compare = greater_than_or_equals } }
		set_variable = { var = ADISCORD_total_development_level value = 4 }
	}
	else_if = {
		limit = { check_variable = { var = ADISCORD_total_development_score value = 18 compare = greater_than_or_equals } }
		set_variable = { var = ADISCORD_total_development_level value = 3 }
	}
	else_if = {
		limit = { check_variable = { var = ADISCORD_total_development_score value = 12 compare = greater_than_or_equals } }
		set_variable = { var = ADISCORD_total_development_level value = 2 }
	}
	else = {
		set_variable = { var = ADISCORD_total_development_level value = 1 }
	}
}
