# A-DISCORD strategic economy MVP.
# The economy is a budget layer. It feeds the existing ADISCORD_economic_development progress and ideas.
ADISCORD_economy_initialize_country = {
	ADISCORD_initialize_society_development_variables = yes
	if = {
		limit = { NOT = { has_variable = ADISCORD_economy_initialized } }
		ADISCORD_economy_set_default_values = yes
		ADISCORD_economy_initialize_from_development = yes
		ADISCORD_economy_set_simulation_tier = yes
		ADISCORD_economy_update_model_and_cycle = yes
		ADISCORD_economy_recalculate_policy_modifiers = yes
		ADISCORD_economy_full_refresh = yes
		ADISCORD_economy_light_update = yes
		ADISCORD_economy_calculate_development_multiplier = yes
		ADISCORD_economy_refresh_spending_ideas = yes
		ADISCORD_economy_update_gui = yes
	}
	else = {
		ADISCORD_economy_initialize_variables = yes
		ADISCORD_economy_set_simulation_tier = yes
	}
}

ADISCORD_economy_set_simulation_tier = {
	if = { limit = { ADISCORD_economy_is_player_tier_country = yes } set_variable = { var = ADISCORD_economy_simulation_tier value = 0 } }
	else_if = { limit = { ADISCORD_economy_is_primary_tier_country = yes } set_variable = { var = ADISCORD_economy_simulation_tier value = 1 } }
	else_if = { limit = { ADISCORD_economy_is_secondary_tier_country = yes } set_variable = { var = ADISCORD_economy_simulation_tier value = 2 } }
	else = { set_variable = { var = ADISCORD_economy_simulation_tier value = 3 } }
}

ADISCORD_economy_calculate_personal_income = {
	# Simple readable tax base: population tax is mostly development/state capacity, not a huge money printer.
	set_variable = { var = ADISCORD_economy_personal_income value = 1.4 }
	if = { limit = { ADISCORD_society_development_at_least_2 = yes } add_to_variable = { var = ADISCORD_economy_personal_income value = 0.3 } }
	if = { limit = { ADISCORD_society_development_at_least_3 = yes } add_to_variable = { var = ADISCORD_economy_personal_income value = 0.4 } }
	if = { limit = { ADISCORD_social_system_development_at_least_3 = yes } add_to_variable = { var = ADISCORD_economy_personal_income value = 0.3 } }
	if = { limit = { ADISCORD_state_development_at_least_3 = yes } add_to_variable = { var = ADISCORD_economy_personal_income value = 0.5 } }
	if = { limit = { ADISCORD_economic_development_at_least_4 = yes } add_to_variable = { var = ADISCORD_economy_personal_income value = 0.4 } }
	if = { limit = { ADISCORD_economy_has_taxation_progressive_brackets = yes } add_to_variable = { var = ADISCORD_economy_personal_income value = 0.8 } }
	if = { limit = { ADISCORD_economy_has_taxation_light_dues = yes } add_to_variable = { var = ADISCORD_economy_personal_income value = -0.5 } }
	if = { limit = { ADISCORD_economy_has_taxation_extraction_quotas = yes } add_to_variable = { var = ADISCORD_economy_personal_income value = 0.7 } }
	clamp_variable = { var = ADISCORD_economy_personal_income min = 0 max = 50 }
}
ADISCORD_economy_calculate_business_income = {
	# Business/taxable services. Stronger than ordinary taxes, weaker than resource rents.
	set_variable = { var = ADISCORD_economy_business_income value = num_of_civilian_factories }
	multiply_variable = { var = ADISCORD_economy_business_income value = 0.075 }
	set_temp_variable = { var = ADISCORD_economy_business_building_temp value = ADISCORD_commercial_office_count }
	multiply_temp_variable = { var = ADISCORD_economy_business_building_temp value = 0.75 }
	add_to_variable = { var = ADISCORD_economy_business_income value = ADISCORD_economy_business_building_temp }
	set_temp_variable = { var = ADISCORD_economy_business_building_temp value = ADISCORD_fiscal_administration_count }
	multiply_temp_variable = { var = ADISCORD_economy_business_building_temp value = 0.15 }
	add_to_variable = { var = ADISCORD_economy_business_income value = ADISCORD_economy_business_building_temp }
	if = { limit = { ADISCORD_economy_has_industrial_civilian_expansion = yes } add_to_variable = { var = ADISCORD_economy_business_income value = 0.5 } }
	if = { limit = { ADISCORD_economy_has_industrial_artisan_markets = yes } multiply_variable = { var = ADISCORD_economy_business_income value = 0.95 } }
	if = { limit = { ADISCORD_economy_has_industrial_military_prioritization = yes } add_to_variable = { var = ADISCORD_economy_business_income value = -0.7 } }
	if = { limit = { ADISCORD_economy_model_is_fragmented = yes } multiply_variable = { var = ADISCORD_economy_business_income value = 0.75 } }
	if = { limit = { ADISCORD_economy_model_is_decentralized_market = yes } multiply_variable = { var = ADISCORD_economy_business_income value = 1.25 } }
	if = { limit = { ADISCORD_economy_model_is_mixed = yes } multiply_variable = { var = ADISCORD_economy_business_income value = 1.10 } }
	if = { limit = { ADISCORD_economy_model_is_state_coordinated = yes } multiply_variable = { var = ADISCORD_economy_business_income value = 0.95 } }

	clamp_variable = { var = ADISCORD_economy_business_income min = 0 max = 100 }
}
ADISCORD_economy_calculate_consumer_goods_income = {
	# Civilian factories are taxable civilian output. This is deliberately below resources/business and below taxes in most states.
	set_variable = { var = ADISCORD_economy_consumer_goods_income value = num_of_civilian_factories }
	multiply_variable = { var = ADISCORD_economy_consumer_goods_income value = 0.035 }
	# Industrial clusters mainly support civilian taxable output. They are not a default money printer for military factories.
	set_temp_variable = { var = ADISCORD_economy_consumer_goods_cluster_temp value = ADISCORD_industrial_cluster_count }
	multiply_temp_variable = { var = ADISCORD_economy_consumer_goods_cluster_temp value = 0.25 }
	add_to_variable = { var = ADISCORD_economy_consumer_goods_income value = ADISCORD_economy_consumer_goods_cluster_temp }
	if = { limit = { ADISCORD_economy_has_industrial_civilian_expansion = yes } add_to_variable = { var = ADISCORD_economy_consumer_goods_income value = 0.6 } }
	if = { limit = { ADISCORD_economy_has_industrial_state_planning = yes } add_to_variable = { var = ADISCORD_economy_consumer_goods_income value = 0.3 } }
	if = { limit = { has_idea = civilian_economy } add_to_variable = { var = ADISCORD_economy_consumer_goods_income value = 0.7 } }
	if = { limit = { has_idea = low_economic_mobilisation } add_to_variable = { var = ADISCORD_economy_consumer_goods_income value = 0.4 } }

ADISCORD_economy_apply_monthly_balance = {
	# V19 TDA-core: one country, one month, one applied balance.
	# If the balance pushes the treasury below zero, the uncovered part becomes automatic debt.
	set_variable = { var = ADISCORD_economy_last_auto_borrowing value = 0 }
	set_variable = { var = ADISCORD_economy_last_uncovered_deficit value = 0 }
	set_variable = { var = ADISCORD_economy_last_monthly_balance_applied value = ADISCORD_economy_monthly_balance }
	set_variable = { var = ADISCORD_economy_treasury_before_tick value = ADISCORD_economy_treasury }

	if = {
		limit = { check_variable = { var = ADISCORD_economy_monthly_balance value = 0 compare = greater_than_or_equals } }
		set_variable = { var = ADISCORD_economy_balance_temp value = ADISCORD_economy_monthly_balance }
		multiply_variable = { var = ADISCORD_economy_balance_temp value = ADISCORD_economy_final_reserve_growth_factor_bp }
		divide_variable = { var = ADISCORD_economy_balance_temp value = 100 }
		add_to_variable = { var = ADISCORD_economy_treasury value = ADISCORD_economy_balance_temp }
		set_variable = { var = ADISCORD_economy_deficit_streak value = 0 }
		add_to_variable = { var = ADISCORD_economy_surplus_streak value = 1 }
		add_to_variable = { var = ADISCORD_economy_deficit_pressure value = -2 }
	}
	else = {
		add_to_variable = { var = ADISCORD_economy_treasury value = ADISCORD_economy_monthly_balance }
		add_to_variable = { var = ADISCORD_economy_deficit_pressure value = 1 }
		add_to_variable = { var = ADISCORD_economy_deficit_streak value = 1 }
		set_variable = { var = ADISCORD_economy_surplus_streak value = 0 }

		# Automatic loan only for uncovered deficit. No huge packages, no inflation spiral.
		if = {
			limit = { check_variable = { var = ADISCORD_economy_treasury value = 0 compare = less_than } }
			set_variable = { var = ADISCORD_economy_uncovered_deficit_temp value = 0 }
			subtract_from_variable = { var = ADISCORD_economy_uncovered_deficit_temp value = ADISCORD_economy_treasury }
			set_variable = { var = ADISCORD_economy_last_uncovered_deficit value = ADISCORD_economy_uncovered_deficit_temp }
			add_to_variable = { var = ADISCORD_economy_debt value = ADISCORD_economy_uncovered_deficit_temp }
			set_variable = { var = ADISCORD_economy_last_auto_borrowing value = ADISCORD_economy_uncovered_deficit_temp }
			set_variable = { var = ADISCORD_economy_treasury value = 0 }
			add_to_variable = { var = ADISCORD_economy_fiscal_stress value = 0.25 }
		}
	}

	set_variable = { var = ADISCORD_economy_treasury_after_tick value = ADISCORD_economy_treasury }
	set_variable = { var = ADISCORD_economy_actual_treasury_delta value = ADISCORD_economy_treasury_after_tick }
	subtract_from_variable = { var = ADISCORD_economy_actual_treasury_delta value = ADISCORD_economy_treasury_before_tick }

	clamp_variable = { var = ADISCORD_economy_deficit_streak min = 0 max = 24 }
	clamp_variable = { var = ADISCORD_economy_surplus_streak min = 0 max = 24 }
	clamp_variable = { var = ADISCORD_economy_deficit_pressure min = 0 max = 100 }
	ADISCORD_economy_calculate_macro_indicators = yes
	ADISCORD_economy_clamp_treasury = yes
}

ADISCORD_economy_update_inflation = {
	# TDA-lite pass: inflation moves slowly and is primarily caused by explicit emission, shocks and crises.
	# Planned-bureaucratic systems suppress visible market inflation; deficits become shortages/overload instead.
	if = {
		limit = { ADISCORD_economy_model_is_planned_bureaucratic = yes }
		set_variable = { var = ADISCORD_economy_emission_pressure value = 0 }
		if = { limit = { check_variable = { var = ADISCORD_economy_inflation value = 0 compare = greater_than } } add_to_variable = { var = ADISCORD_economy_inflation value = -2 } }
		if = { limit = { check_variable = { var = ADISCORD_economy_money_printing_level value = 1 compare = greater_than_or_equals } } add_to_variable = { var = ADISCORD_economy_stretched_score value = ADISCORD_economy_money_printing_level } }
		if = { limit = { check_variable = { var = ADISCORD_economy_deficit_pressure value = 20 compare = greater_than_or_equals } } add_to_variable = { var = ADISCORD_economy_stretched_score value = 1 } }
	}
	else = {
		set_variable = { var = ADISCORD_economy_emission_pressure value = ADISCORD_economy_money_printing_level }
		multiply_variable = { var = ADISCORD_economy_emission_pressure value = 2 }
		multiply_variable = { var = ADISCORD_economy_emission_pressure value = 100 }
		divide_variable = { var = ADISCORD_economy_emission_pressure value = ADISCORD_economy_final_money_printing_efficiency_factor_bp }
		set_variable = { var = ADISCORD_economy_inflation_pressure_temp value = ADISCORD_economy_emission_pressure }
		set_variable = { var = ADISCORD_economy_inflation_component_temp value = ADISCORD_economy_deficit_pressure }
		divide_variable = { var = ADISCORD_economy_inflation_component_temp value = 40 }
		add_to_variable = { var = ADISCORD_economy_inflation_pressure_temp value = ADISCORD_economy_inflation_component_temp }
		set_variable = { var = ADISCORD_economy_inflation_component_temp value = ADISCORD_economy_price_shock }
		divide_variable = { var = ADISCORD_economy_inflation_component_temp value = 5 }
		add_to_variable = { var = ADISCORD_economy_inflation_pressure_temp value = ADISCORD_economy_inflation_component_temp }
		set_variable = { var = ADISCORD_economy_inflation_component_temp value = ADISCORD_economy_debt_crisis_level }
		multiply_variable = { var = ADISCORD_economy_inflation_component_temp value = 0.5 }
		add_to_variable = { var = ADISCORD_economy_inflation_pressure_temp value = ADISCORD_economy_inflation_component_temp }
		if = { limit = { has_war = yes } add_to_variable = { var = ADISCORD_economy_inflation_pressure_temp value = 0.25 } }
		if = { limit = { ADISCORD_economy_model_is_mobilization = yes } add_to_variable = { var = ADISCORD_economy_inflation_pressure_temp value = 0.5 } }
		multiply_variable = { var = ADISCORD_economy_inflation_pressure_temp value = ADISCORD_economy_final_inflation_pressure_factor_bp }
		divide_variable = { var = ADISCORD_economy_inflation_pressure_temp value = 100 }
		multiply_variable = { var = ADISCORD_economy_inflation_pressure_temp value = 100 }
		divide_variable = { var = ADISCORD_economy_inflation_pressure_temp value = ADISCORD_economy_final_price_stability_factor_bp }
		set_variable = { var = ADISCORD_economy_inflation_delta_temp value = ADISCORD_economy_inflation_pressure_temp }
		divide_variable = { var = ADISCORD_economy_inflation_delta_temp value = 6 }
		if = { limit = { check_variable = { var = ADISCORD_economy_monthly_balance value = 0 compare = greater_than_or_equals } } add_to_variable = { var = ADISCORD_economy_inflation_delta_temp value = -0.5 } }
		if = { limit = { check_variable = { var = ADISCORD_economy_surplus_streak value = 3 compare = greater_than_or_equals } } add_to_variable = { var = ADISCORD_economy_inflation_delta_temp value = -0.5 } }
		if = { limit = { check_variable = { var = ADISCORD_economy_money_printing_level value = 0 compare = less_than_or_equals } } add_to_variable = { var = ADISCORD_economy_inflation_delta_temp value = -0.25 } }
		clamp_variable = { var = ADISCORD_economy_inflation_delta_temp min = -1.0 max = 1.5 }
		add_to_variable = { var = ADISCORD_economy_inflation value = ADISCORD_economy_inflation_delta_temp }
	}
	if = { limit = { check_variable = { var = ADISCORD_economy_price_shock value = 0 compare = greater_than } } add_to_variable = { var = ADISCORD_economy_price_shock value = -0.5 } }
	if = { limit = { check_variable = { var = ADISCORD_economy_deficit_pressure value = 0 compare = greater_than } check_variable = { var = ADISCORD_economy_monthly_balance value = 0 compare = greater_than_or_equals } } add_to_variable = { var = ADISCORD_economy_deficit_pressure value = -2 } }
	clamp_variable = { var = ADISCORD_economy_inflation min = 0 max = 100 }
	clamp_variable = { var = ADISCORD_economy_emission_pressure min = 0 max = 100 }
	clamp_variable = { var = ADISCORD_economy_price_shock min = 0 max = 100 }
	clamp_variable = { var = ADISCORD_economy_deficit_pressure min = 0 max = 100 }
}


ADISCORD_economy_update_macro_confidence = {
	set_variable = { var = ADISCORD_economy_investment_confidence value = 50 }
	set_variable = { var = ADISCORD_economy_confidence_temp value = ADISCORD_economy_creditworthiness }
	divide_variable = { var = ADISCORD_economy_confidence_temp value = 2 }
	add_to_variable = { var = ADISCORD_economy_investment_confidence value = ADISCORD_economy_confidence_temp }
	set_variable = { var = ADISCORD_economy_confidence_temp value = ADISCORD_economy_inflation }
	divide_variable = { var = ADISCORD_economy_confidence_temp value = 2 }
	subtract_from_variable = { var = ADISCORD_economy_investment_confidence value = ADISCORD_economy_confidence_temp }
	set_variable = { var = ADISCORD_economy_confidence_temp value = ADISCORD_economy_fiscal_stress }
	divide_variable = { var = ADISCORD_economy_confidence_temp value = 2 }
	subtract_from_variable = { var = ADISCORD_economy_investment_confidence value = ADISCORD_economy_confidence_temp }
	set_variable = { var = ADISCORD_economy_confidence_temp value = ADISCORD_economy_debt_ratio }
	divide_variable = { var = ADISCORD_economy_confidence_temp value = 4 }
	subtract_from_variable = { var = ADISCORD_economy_investment_confidence value = ADISCORD_economy_confidence_temp }
	multiply_variable = { var = ADISCORD_economy_investment_confidence value = ADISCORD_economy_final_investment_confidence_factor_bp }
	divide_variable = { var = ADISCORD_economy_investment_confidence value = 100 }
	set_variable = { var = ADISCORD_economy_state_financial_control value = 50 }
	set_variable = { var = ADISCORD_economy_control_temp value = ADISCORD_state_development_level }
	multiply_variable = { var = ADISCORD_economy_control_temp value = 8 }
	add_to_variable = { var = ADISCORD_economy_state_financial_control value = ADISCORD_economy_control_temp }
	set_variable = { var = ADISCORD_economy_control_temp value = ADISCORD_fiscal_administration_count }
	multiply_variable = { var = ADISCORD_economy_control_temp value = 4 }
	add_to_variable = { var = ADISCORD_economy_state_financial_control value = ADISCORD_economy_control_temp }
	subtract_from_variable = { var = ADISCORD_economy_state_financial_control value = ADISCORD_economy_fiscal_stress }
	set_variable = { var = ADISCORD_economy_control_temp value = ADISCORD_economy_stretched_level }
	multiply_variable = { var = ADISCORD_economy_control_temp value = 5 }
	subtract_from_variable = { var = ADISCORD_economy_state_financial_control value = ADISCORD_economy_control_temp }
	if = { limit = { ADISCORD_economy_model_is_state_coordinated = yes } add_to_variable = { var = ADISCORD_economy_state_financial_control value = 8 } }
	if = { limit = { ADISCORD_economy_model_is_planned_bureaucratic = yes } add_to_variable = { var = ADISCORD_economy_state_financial_control value = 10 } }
	if = { limit = { ADISCORD_economy_model_is_fragmented = yes } add_to_variable = { var = ADISCORD_economy_state_financial_control value = -15 } }
	set_variable = { var = ADISCORD_economy_austerity_level value = 0 }
	if = { limit = { check_variable = { var = ADISCORD_economy_fiscal_stress value = 35 compare = greater_than_or_equals } } set_variable = { var = ADISCORD_economy_austerity_level value = 1 } }
	if = { limit = { check_variable = { var = ADISCORD_economy_fiscal_stress value = 55 compare = greater_than_or_equals } } set_variable = { var = ADISCORD_economy_austerity_level value = 2 } }
	if = { limit = { check_variable = { var = ADISCORD_economy_fiscal_stress value = 75 compare = greater_than_or_equals } } set_variable = { var = ADISCORD_economy_austerity_level value = 3 } }
	if = { limit = { check_variable = { var = ADISCORD_economy_debt_crisis_level value = 4 compare = greater_than_or_equals } } set_variable = { var = ADISCORD_economy_austerity_level value = 4 } }
	clamp_variable = { var = ADISCORD_economy_investment_confidence min = 0 max = 100 }
	clamp_variable = { var = ADISCORD_economy_state_financial_control min = 0 max = 100 }
	clamp_variable = { var = ADISCORD_economy_austerity_level min = 0 max = 4 }
}

ADISCORD_economy_clamp_treasury = {
	if = { limit = { check_variable = { var = ADISCORD_economy_treasury value = 0 compare = less_than } } set_variable = { var = ADISCORD_economy_treasury value = 0 } }
	if = {
		limit = { check_variable = { var = ADISCORD_economy_treasury value = ADISCORD_economy_treasury_cap compare = greater_than } }
		set_variable = { var = ADISCORD_economy_treasury value = ADISCORD_economy_treasury_cap }
	}
}

	multiply_variable = { var = ADISCORD_economy_monthly_development_gain value = ADISCORD_economy_monthly_development_multiplier }
	divide_variable = { var = ADISCORD_economy_monthly_development_gain value = 100 }
	if = {
		limit = {
			NOT = { ADISCORD_economy_can_upgrade_economic_development = yes }
			check_variable = { var = ADISCORD_economic_development_progress value = 95 compare = greater_than_or_equals }
		}
		set_variable = { var = ADISCORD_economy_monthly_development_gain value = 0 }
	}
	set_variable = { var = ADISCORD_economy_development_progress value = ADISCORD_economic_development_progress }
}

ADISCORD_economy_apply_development_progress = {
	add_to_variable = { var = ADISCORD_economic_development_progress value = ADISCORD_economy_monthly_development_gain }
	ADISCORD_economy_check_economic_development_upgrade = yes
	set_variable = { var = ADISCORD_economy_development_progress value = ADISCORD_economic_development_progress }
}

ADISCORD_economy_check_economic_development_upgrade = {
	if = {
		limit = { check_variable = { var = ADISCORD_economic_development_progress value = 100 compare = greater_than_or_equals } }
		if = {
			limit = { ADISCORD_economy_can_upgrade_economic_development = yes }
			set_variable = { var = ADISCORD_economic_development_progress value = 0 }
			ADISCORD_increase_economic_development = yes
		}
		else = {
			set_variable = { var = ADISCORD_economic_development_progress value = 99 }
		}
	}
	ADISCORD_economy_refresh_existing_economic_development_idea = yes
}

ADISCORD_economy_refresh_existing_economic_development_idea = {
	remove_ideas = ADISCORD_economic_development_1
	remove_ideas = ADISCORD_economic_development_2
	remove_ideas = ADISCORD_economic_development_3
	remove_ideas = ADISCORD_economic_development_4
	remove_ideas = ADISCORD_economic_development_5
	if = { limit = { ADISCORD_economic_development_at_most_1 = yes } add_ideas = ADISCORD_economic_development_1 }
	else_if = { limit = { ADISCORD_economic_development_exact_2 = yes } add_ideas = ADISCORD_economic_development_2 }
	else_if = { limit = { ADISCORD_economic_development_exact_3 = yes } add_ideas = ADISCORD_economic_development_3 }
	else_if = { limit = { ADISCORD_economic_development_exact_4 = yes } add_ideas = ADISCORD_economic_development_4 }
	else = { add_ideas = ADISCORD_economic_development_5 }
}

ADISCORD_economy_update_war_fatigue = {
	set_variable = { var = ADISCORD_economy_war_fatigue_before_temp value = ADISCORD_economy_war_fatigue_score }
	if = { limit = { has_war = yes } add_to_variable = { var = ADISCORD_economy_war_fatigue_score value = 2 } }
	else = { add_to_variable = { var = ADISCORD_economy_war_fatigue_score value = -1 } }
	if = { limit = { casualties_k > 50 } add_to_variable = { var = ADISCORD_economy_war_fatigue_score value = 1 } }
	if = { limit = { casualties_k > 100 } add_to_variable = { var = ADISCORD_economy_war_fatigue_score value = 2 } }
	if = { limit = { casualties_k > 250 } add_to_variable = { var = ADISCORD_economy_war_fatigue_score value = 3 } }
	if = { limit = { check_variable = { var = ADISCORD_economy_army_spending_mode value = 5 compare = greater_than_or_equals } has_war = yes } add_to_variable = { var = ADISCORD_economy_war_fatigue_score value = 2 } }
	if = { limit = { ADISCORD_economy_has_military_total_defense_grid = yes } add_to_variable = { var = ADISCORD_economy_war_fatigue_score value = 1 } }
	if = { limit = { ADISCORD_economy_has_labor_mobilized_labor = yes } add_to_variable = { var = ADISCORD_economy_war_fatigue_score value = 1 } }
	if = { limit = { ADISCORD_economy_has_taxation_extraction_quotas = yes } add_to_variable = { var = ADISCORD_economy_war_fatigue_score value = 1 } }
	set_variable = { var = ADISCORD_economy_war_fatigue_delta_temp value = ADISCORD_economy_war_fatigue_score }
	subtract_from_variable = { var = ADISCORD_economy_war_fatigue_delta_temp value = ADISCORD_economy_war_fatigue_before_temp }
	if = {
		limit = { check_variable = { var = ADISCORD_economy_war_fatigue_delta_temp value = 0 compare = greater_than } }
		multiply_variable = { var = ADISCORD_economy_war_fatigue_delta_temp value = ADISCORD_economy_final_war_fatigue_gain_factor_bp }
		divide_variable = { var = ADISCORD_economy_war_fatigue_delta_temp value = 100 }
		set_variable = { var = ADISCORD_economy_war_fatigue_score value = ADISCORD_economy_war_fatigue_before_temp }
		add_to_variable = { var = ADISCORD_economy_war_fatigue_score value = ADISCORD_economy_war_fatigue_delta_temp }
	}
	clamp_variable = { var = ADISCORD_economy_war_fatigue_score min = 0 max = 100 }
	if = { limit = { check_variable = { var = ADISCORD_economy_war_fatigue_score value = 75 compare = greater_than_or_equals } } set_variable = { var = ADISCORD_economy_war_fatigue_level value = 4 } }
	else_if = { limit = { check_variable = { var = ADISCORD_economy_war_fatigue_score value = 50 compare = greater_than_or_equals } } set_variable = { var = ADISCORD_economy_war_fatigue_level value = 3 } }
	else_if = { limit = { check_variable = { var = ADISCORD_economy_war_fatigue_score value = 30 compare = greater_than_or_equals } } set_variable = { var = ADISCORD_economy_war_fatigue_level value = 2 } }
	else_if = { limit = { check_variable = { var = ADISCORD_economy_war_fatigue_score value = 15 compare = greater_than_or_equals } } set_variable = { var = ADISCORD_economy_war_fatigue_level value = 1 } }
	else = { set_variable = { var = ADISCORD_economy_war_fatigue_level value = 0 } }
}

ADISCORD_economy_update_model_and_cycle = {
	# V7: explicit economic-system laws are authoritative when present.
	# If an old save has no ADISCORD_economic_system_* law, fall back to the old inferred model logic.
	set_variable = { var = ADISCORD_economy_model value = 2 }

	if = { limit = { has_idea = ADISCORD_economic_system_agrarian } set_variable = { var = ADISCORD_economy_model value = 0 } }
	if = { limit = { has_idea = ADISCORD_economic_system_industrializing } set_variable = { var = ADISCORD_economy_model value = 2 } }
	if = { limit = { has_idea = ADISCORD_economic_system_free_market } set_variable = { var = ADISCORD_economy_model value = 1 } }
	if = { limit = { has_idea = ADISCORD_economic_system_mixed } set_variable = { var = ADISCORD_economy_model value = 2 } }
	if = { limit = { has_idea = ADISCORD_economic_system_state_coordinated } set_variable = { var = ADISCORD_economy_model value = 3 } }
	if = { limit = { has_idea = ADISCORD_economic_system_planned_bureaucratic } set_variable = { var = ADISCORD_economy_model value = 4 } }
	if = { limit = { has_idea = ADISCORD_economic_system_mobilization } set_variable = { var = ADISCORD_economy_model value = 5 } }
	if = { limit = { has_idea = ADISCORD_economic_system_oligarchic_clan } set_variable = { var = ADISCORD_economy_model value = 6 } }
	if = { limit = { has_idea = ADISCORD_economic_system_technocratic } set_variable = { var = ADISCORD_economy_model value = 7 } }

	if = {
		limit = {
			NOT = {
				OR = {
					has_idea = ADISCORD_economic_system_agrarian
					has_idea = ADISCORD_economic_system_industrializing
					has_idea = ADISCORD_economic_system_free_market
					has_idea = ADISCORD_economic_system_mixed
					has_idea = ADISCORD_economic_system_state_coordinated
					has_idea = ADISCORD_economic_system_planned_bureaucratic
					has_idea = ADISCORD_economic_system_mobilization
					has_idea = ADISCORD_economic_system_oligarchic_clan
					has_idea = ADISCORD_economic_system_technocratic
				}
			}
		}

		set_variable = { var = ADISCORD_economy_model value = 2 }
		if = { limit = { ADISCORD_economy_has_industrial_artisan_markets = yes ADISCORD_state_development_at_most_2 = yes } set_variable = { var = ADISCORD_economy_model value = 0 } }
		if = { limit = { ADISCORD_economy_has_taxation_light_dues = yes ADISCORD_economy_has_labor_loose_contracts = yes } set_variable = { var = ADISCORD_economy_model value = 1 } }
		if = { limit = { ADISCORD_economy_has_industrial_state_planning = yes ADISCORD_state_development_at_least_3 = yes } set_variable = { var = ADISCORD_economy_model value = 3 } }
		if = { limit = { ADISCORD_economy_has_industrial_state_planning = yes ADISCORD_state_development_at_least_4 = yes } set_variable = { var = ADISCORD_economy_model value = 4 } }
		if = { limit = { ADISCORD_economy_has_taxation_extraction_quotas = yes ADISCORD_economy_has_industrial_military_prioritization = yes } set_variable = { var = ADISCORD_economy_model value = 5 } }
		if = { limit = { ADISCORD_state_development_at_most_1 = yes ADISCORD_economy_has_taxation_industrial_tariffs = yes } set_variable = { var = ADISCORD_economy_model value = 6 } }
		if = { limit = { has_government = technocracy ADISCORD_economy_has_labor_technocratic_work_norms = yes } set_variable = { var = ADISCORD_economy_model value = 7 } }
	}

	set_variable = { var = ADISCORD_economy_cycle_phase value = 1 }
	if = { limit = { check_variable = { var = ADISCORD_economy_monthly_balance value = 0 compare = less_than } } set_variable = { var = ADISCORD_economy_cycle_phase value = 0 } }
	if = { limit = { check_variable = { var = ADISCORD_economy_monthly_balance value = 5 compare = greater_than_or_equals } check_variable = { var = ADISCORD_economy_stretched_level value = 2 compare = less_than } } set_variable = { var = ADISCORD_economy_cycle_phase value = 2 } }
	if = { limit = { check_variable = { var = ADISCORD_economy_inflation value = 30 compare = greater_than_or_equals } } set_variable = { var = ADISCORD_economy_cycle_phase value = 3 } }
	if = { limit = { check_variable = { var = ADISCORD_economy_stretched_level value = 3 compare = greater_than_or_equals } } set_variable = { var = ADISCORD_economy_cycle_phase value = 4 } }
	if = { limit = { has_war = yes OR = { ADISCORD_economy_has_taxation_extraction_quotas = yes ADISCORD_economy_has_industrial_military_prioritization = yes has_idea = ADISCORD_economic_system_mobilization } } set_variable = { var = ADISCORD_economy_cycle_phase value = 5 } }
	# State-planned economies do not use market business cycles in the dashboard.
	if = { limit = { ADISCORD_economy_model_is_planned_bureaucratic = yes } set_variable = { var = ADISCORD_economy_cycle_phase value = 6 } }
	if = { limit = { ADISCORD_economy_model_is_mobilization = yes } set_variable = { var = ADISCORD_economy_cycle_phase value = 7 } }
}
