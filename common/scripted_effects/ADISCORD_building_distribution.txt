# common/scripted_effects/ADISCORD_building_distribution.txt

ADISCORD_distribute_starting_buildings_by_country = {
	if = {
		limit = { NOT = { has_country_flag = ADISCORD_starting_buildings_distributed } }

		# STP: developed/service-heavy start.
		if = {
			limit = { tag = STP }
			capital_scope = {
				if = {
					limit = {
						free_building_slots = {
							building = ADISCORD_fiscal_administration
							size > 0
							include_locked = yes
						}
					}
					add_building_construction = {
						type = ADISCORD_fiscal_administration
						level = 1
						instant_build = yes
					}
				}
				if = {
					limit = {
						free_building_slots = {
							building = ADISCORD_commercial_office
							size > 0
							include_locked = yes
						}
					}
					add_building_construction = {
						type = ADISCORD_commercial_office
						level = 1
						instant_build = yes
					}
				}
			}
			random_owned_controlled_state = {
				limit = {
					is_core_of = ROOT
					free_building_slots = {
						building = ADISCORD_commercial_office
						size > 0
						include_locked = yes
					}
				}
				add_building_construction = {
					type = ADISCORD_commercial_office
					level = 1
					instant_build = yes
				}
			}
			random_owned_controlled_state = {
				limit = {
					is_core_of = ROOT
					free_building_slots = {
						building = ADISCORD_logistics_hub
						size > 0
						include_locked = yes
					}
				}
				add_building_construction = {
					type = ADISCORD_logistics_hub
					level = 1
					instant_build = yes
				}
			}
			random_owned_controlled_state = {
				limit = {
					is_core_of = ROOT
					free_building_slots = {
						building = ADISCORD_technical_institute
						size > 0
						include_locked = yes
					}
				}
				add_building_construction = {
					type = ADISCORD_technical_institute
					level = 1
					instant_build = yes
				}
			}
		}

		# WRK: industrial base.
		if = {
			limit = { tag = WRK }
			capital_scope = {
				if = {
					limit = {
						free_building_slots = {
							building = ADISCORD_fiscal_administration
							size > 0
							include_locked = yes
						}
					}
					add_building_construction = {
						type = ADISCORD_fiscal_administration
						level = 1
						instant_build = yes
					}
				}
			}
			random_owned_controlled_state = {
				limit = {
					is_core_of = ROOT
					free_building_slots = {
						building = ADISCORD_industrial_cluster
						size > 0
						include_locked = yes
					}
				}
				add_building_construction = {
					type = ADISCORD_industrial_cluster
					level = 1
					instant_build = yes
				}
			}
			random_owned_controlled_state = {
				limit = {
					is_core_of = ROOT
					free_building_slots = {
						building = ADISCORD_industrial_cluster
						size > 0
						include_locked = yes
					}
				}
				add_building_construction = {
					type = ADISCORD_industrial_cluster
					level = 1
					instant_build = yes
				}
			}
			random_owned_controlled_state = {
				limit = {
					is_core_of = ROOT
					free_building_slots = {
						building = ADISCORD_logistics_hub
						size > 0
						include_locked = yes
					}
				}
				add_building_construction = {
					type = ADISCORD_logistics_hub
					level = 1
					instant_build = yes
				}
			}
		}

		# VAL: mixed industry / logistics / commerce.
		if = {
			limit = { tag = VAL }
			random_owned_controlled_state = {
				limit = {
					is_core_of = ROOT
					free_building_slots = {
						building = ADISCORD_industrial_cluster
						size > 0
						include_locked = yes
					}
				}
				add_building_construction = {
					type = ADISCORD_industrial_cluster
					level = 1
					instant_build = yes
				}
			}
			random_owned_controlled_state = {
				limit = {
					is_core_of = ROOT
					free_building_slots = {
						building = ADISCORD_industrial_cluster
						size > 0
						include_locked = yes
					}
				}
				add_building_construction = {
					type = ADISCORD_industrial_cluster
					level = 1
					instant_build = yes
				}
			}
			random_owned_controlled_state = {
				limit = {
					is_core_of = ROOT
					free_building_slots = {
						building = ADISCORD_logistics_hub
						size > 0
						include_locked = yes
					}
				}
				add_building_construction = {
					type = ADISCORD_logistics_hub
					level = 1
					instant_build = yes
				}
			}
			random_owned_controlled_state = {
				limit = {
					is_core_of = ROOT
					free_building_slots = {
						building = ADISCORD_commercial_office
						size > 0
						include_locked = yes
					}
				}
				add_building_construction = {
					type = ADISCORD_commercial_office
					level = 1
					instant_build = yes
				}
			}
		}

		set_country_flag = ADISCORD_starting_buildings_distributed
	}
}
