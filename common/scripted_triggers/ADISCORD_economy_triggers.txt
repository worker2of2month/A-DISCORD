# A-DISCORD economy scripted triggers.
# These reuse the existing development country lists instead of creating a parallel playable-country system.

ADISCORD_economy_is_player_tier_country = {
	is_ai = no
}

ADISCORD_economy_is_primary_tier_country = {
	OR = {
		ADISCORD_is_playable_development_country = yes
		is_major = yes
		is_faction_leader = yes
		has_country_flag = ADISCORD_economy_important_country
		any_enemy_country = { is_ai = no }
	}
}

ADISCORD_economy_is_secondary_tier_country = {
	is_ai = yes
	OR = {
		ADISCORD_is_secondary_development_country = yes
		has_country_flag = ADISCORD_economy_secondary_country
	}
	NOT = { ADISCORD_economy_is_primary_tier_country = yes }
}

ADISCORD_economy_is_dormant_tier_country = {
	is_ai = yes
	NOT = {
		OR = {
			ADISCORD_economy_is_primary_tier_country = yes
			ADISCORD_economy_is_secondary_tier_country = yes
		}
	}
}

ADISCORD_economy_is_dormant = {
	ADISCORD_economy_is_dormant_tier_country = yes
}

ADISCORD_economy_should_monthly_update = {
	OR = {
		ADISCORD_economy_is_player_tier_country = yes
		ADISCORD_economy_is_primary_tier_country = yes

}

ADISCORD_economy_should_yearly_update = {
