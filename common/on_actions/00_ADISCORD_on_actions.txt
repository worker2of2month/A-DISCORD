####Country Development

on_actions = {
	on_startup = {
		effect = {
			create_entity = { ##poluted zone
			    entity = ADISCORD_global_entity
			    id = 1
			    x = 42
			    y = 21
			    z = -100.000
			    province = 16531
			    state = 42
			    rotation = 1.2
			    scale = 0.03
			    min_zoom = 5555.0
			    #visible = scripted_trigger_name
			}
			create_entity = {
			    entity = ADISCORD_global_entity
			    id = 2
			    x = 42
			    y = 21
			    z = -100.000
			    province = 12863
			    state = 42
			    rotation = 1.2
			    scale = 0.02
			    min_zoom = 5555.0
			    #visible = scripted_trigger_name
			}
			#create_entity = { ##clouds
			#    entity = ADISCORD_global_entity_2
			#    id = 2
			#    x = 42
			#    y = 21
			#    z = -100.000
			#    province = 6713
			#    state = 42
			#    rotation = 1.2
			#    scale = 1.02
			#    min_zoom = 5555.0
			#    #visible = scripted_trigger_name
			#}
			every_country = {
				# The scenario starts in 2160.  Old recovered knowledge should not
				# consume the first decades of every country's research queue.
				ADISCORD_grant_2150_technology_baseline = yes
			}
			every_country = {
				limit = { OR = { ADISCORD_is_playable_development_country = yes is_ai = no } }
				ADISCORD_initialize_default_country_development = yes
			}
			STP = {
				# Starting society development levels for StepanLand.
				ADISCORD_set_society_development_2 = yes
				ADISCORD_set_social_system_development_2 = yes
				ADISCORD_set_army_development_2 = yes
				ADISCORD_set_cultural_development_1 = yes
				ADISCORD_set_state_development_3 = yes
				ADISCORD_set_economic_development_4 = yes
				ADISCORD_initialize_society_development_variables = yes
				ADISCORD_update_total_country_development = yes
			}
			VAL = {
				# Starting society development levels for ValeraLand.
				# Contract state: strong mercenary army, strong arms economy,
				# weak social base, harsh culture, decent but not excellent state apparatus.
				ADISCORD_set_society_development_2 = yes
				ADISCORD_set_social_system_development_1 = yes
				ADISCORD_set_army_development_4 = yes
				ADISCORD_set_cultural_development_2 = yes
				ADISCORD_set_state_development_2 = yes
				ADISCORD_set_economic_development_4 = yes
				ADISCORD_initialize_society_development_variables = yes
				ADISCORD_update_total_country_development = yes
			}
			every_country = {
				limit = {
					OR = {
						ADISCORD_economy_should_monthly_update = yes
						ADISCORD_economy_should_yearly_update = yes
					}
				}
				ADISCORD_economy_initialize_country = yes
				# V19 TDA-core: disabled hidden startup economic pseudo-building distribution.
				# ADISCORD_distribute_starting_buildings_by_country = yes
			}
		}
	}
	on_monthly = {
		effect = {
			# HOI4 monthly pulse is already country-scoped. Do not every_country here.
			if = {
				limit = { ADISCORD_economy_should_monthly_update = yes }
				ADISCORD_economy_monthly_update = yes
			}
			if = {
				limit = { OR = { ADISCORD_is_playable_development_country = yes is_ai = no } }
				ADISCORD_tick_all_society_development_monthly = yes
			}
		}
	}
	on_weekly = {
		effect = {
			# Country-scoped weekly GUI refresh, TDA-style.
			if = {
				limit = { is_ai = no }
				ADISCORD_economy_weekly_player_refresh = yes
			}
		}
	}

	on_yearly = {
		effect = {
			# Country-scoped yearly update. The trigger decides whether this country is secondary-tier.
			if = {
				limit = { ADISCORD_economy_should_yearly_update = yes }
				ADISCORD_economy_yearly_update = yes
			}
			if = {
				limit = {
					ADISCORD_is_secondary_development_country = yes
					is_ai = yes
				}
				ADISCORD_tick_all_society_development_yearly = yes
			}
		}
	}
}
