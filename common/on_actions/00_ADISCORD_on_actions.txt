	on_monthly = {
		effect = {
			every_country = {
				limit = { OR = { ADISCORD_is_playable_development_country = yes is_ai = no } }
				ADISCORD_tick_all_society_development_monthly = yes
			}
		}
	}
	on_yearly = {
		effect = {
			every_country = {
				limit = {
